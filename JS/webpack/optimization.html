    • 多页面抽离公共代码

      // splitChunks 多页面抽取公共模块
entry: {
  index: './src/index.js', //多页抽取公共模块
  other: './src/other.js'
},
optimization: {
  splitChunks: {
    cacheGroups: {
      common: {
        chunks: 'initial' // 模式
        minSize: 0, //大小
        minChunks: 2, //至少两次引用
      },
      vendor: {
        test: /node_modules/,
        chunks: 'initial',
        minSzie: 0,
        minChunks: 2,
        priority: 1, //权重，优先抽离第三方模块
      }
    }
  }
}
